/** * تم تحديث منطق ترتيب الصفوف (V1 - Locked) * الترتيب: 1. إيجار | 2. معارف | 3. رصيد مدور | 4. الدفعات (حسب التاريخ) */ function render(prevBal, transactions) { const container = document.getElementById('years_container'); container.innerHTML = ''; const grouped = {}; transactions.forEach(t => { const dateObj = new Date(t.date); let y = dateObj.getFullYear(); const desc = (t.description || '').trim(); if (isCarryRow(desc)) { y = y + 1; } if (!grouped[y]) grouped[y] = []; grouped[y].push(t); }); const years = Object.keys(grouped).map(Number).sort((a, b) => a - b); let carryBalance = prevBal; years.forEach(year => { const rows = grouped[year] || []; // المنطق المحدث لترتيب صفوف الافتتاح (Opening Rows) const openingRows = rows .filter(r => isRentRow(r.description || '') || isFeesRow(r.description || '') || isCarryRow(r.description || '') ) .sort((a, b) => { const descA = (a.description || '').trim(); const descB = (b.description || '').trim(); const getOpeningOrder = (desc) => { if (isRentRow(desc)) return 1; if (isFeesRow(desc)) return 2; if (isCarryRow(desc)) return 3; return 99; }; const orderA = getOpeningOrder(descA); const orderB = getOpeningOrder(descB); if (orderA !== orderB) return orderA - orderB; return new Date(a.date) - new Date(b.date); }); // ترتيب صفوف الدفعات والعمليات الأخرى حسب التاريخ const paymentRows = rows .filter(r => !isRentRow(r.description || '') && !isFeesRow(r.description || '') && !isCarryRow(r.description || '') ) .sort((a, b) => new Date(a.date) - new Date(b.date)); const yearRows = [ ...openingRows, ...paymentRows ]; const hasCarryRow = yearRows.some(r => isCarryRow(r.description || '')); let items = calculateRunningBalance( yearRows, hasCarryRow ? 0 : carryBalance ); items = items.map((item, idx) => { const desc = (item.description || '').trim(); return { ...item, isCarry: isCarryRow(desc), isMain: isRentRow(desc) || isFeesRow(desc), isLast: idx === items.length - 1 }; }); const finalBalance = items.length ? items[items.length - 1].balance : carryBalance; const totalCredit = items.reduce((sum, r) => sum + toNumber(r.credit), 0); let status = { text: 'متأخر', className: 'status-late', chip: 'late' }; if (finalBalance <= 0) { status = { text: 'مسدد', className: 'status-paid', chip: 'paid' }; } else if (totalCredit > 0) { status = { text: 'جزئي', className: 'status-partial', chip: 'partial' }; } const card = document.createElement('div'); card.className = `year-card ${status.className}`; card.innerHTML = `
${status.text}
${year}
${items.map(item => ` `).join('')}
التاريخ مدين دائن الرصيد البيان سند
${formatShortDate(item.date)} ${item.debit ? formatCurrency(item.debit) : ''} ${item.credit ? formatCurrency(item.credit) : ''} ${formatCurrency(item.balance)} ${item.description || ''} ${item.receipt_no || '—'}
دائن ${formatCurrency(totalCredit)}
رصيد ${formatCurrency(finalBalance)}
`; container.appendChild(card); carryBalance = finalBalance; }); }
